    window.filtrarDiccionario = () => {
        const query = document.getElementById('searchDiccionario').value.toLowerCase().trim();
        const container = document.getElementById('diccionarioResult');
        container.innerHTML = '';

        if (!query) {
            container.innerHTML = `
                <div class="empty-state">
                    <span class="material-icons-round" style="font-size: 4rem; color: #ddd;">search</span>
                    <p>Escribe una palabra para buscar su significado legal.</p>
                </div>`;
            return;
        }

        const matches = Object.keys(DICCIONARIO_JURIDICO).filter(k => k.includes(query));

        if (matches.length === 0) {
            container.innerHTML = `
                <div class="empty-state">
                    <span class="material-icons-round" style="font-size: 4rem; color: #ddd;">sentiment_dissatisfied</span>
                    <p>No se encontró el término "${query}". <br> Prueba buscando en la RAE.</p>
                </div>`;
            return;
        }

        matches.forEach(key => {
            const item = document.createElement('div');
            item.className = 'diccionario-item';
            item.innerHTML = `
                <h4>${key}</h4>
                <p>${DICCIONARIO_JURIDICO[key]}</p>
            `;
            container.appendChild(item);
        });
    };

    window.buscarEnRAE = () => {

        } else {
            window.open(`https://dpej.rae.es/`, '_blank');
        }
    };

    window.limpiarCorrector = () => {

    window.verificarTexto = () => {
        const text = document.getElementById('correctorText').value;
        const resultArea = document.getElementById('correctorResults');
        const sugerenciasList = document.getElementById('sugerenciasList');
        const stats = document.getElementById('correctorStats');

        if (!text.trim()) return alert('Por favor, ingresa un texto para verificar.');

        const words = text.split(/[\s,.;:!?()"]+/).filter(w => w.length > 3);
        const uniqueWords = [...new Set(words)];
        const errors = [];

        uniqueWords.forEach(word => {
            const cleanWord = word.toLowerCase().trim();
            // Simple logic: if contains non-spanish typical chars or is rare but sounds juridical and misspelled
            // We check against our local dictionary for "Critical Legal Terms"
            // For general spelling, we rely on the browser's red underline (spellcheck=true)
            // But here we provide specific legal suggestions

            // Check for common typos in legal words
            for (let term in DICCIONARIO_JURIDICO) {
                if (levenshteinDistance(cleanWord, term) === 1 && cleanWord !== term) {
                    errors.push({ original: word, suggestion: term });
                    break;
                }
            }
        });

        stats.textContent = `${words.length} palabras analizadas | ${errors.length} sugerencias legales`;

        if (errors.length > 0) {
            resultArea.style.display = 'block';
            sugerenciasList.innerHTML = '';
            errors.forEach(err => {
                const item = document.createElement('div');
